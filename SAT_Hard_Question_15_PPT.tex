% --- Question 1 ---
\begin{frame}{Question 1}
\small
Which of the following ensures that the quadratic equation
\[
ax^{2} - 4x + c = 0
\]
has at least one real root?
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] \(a > 0\)
    \item[B.] \(a = 0\)
    \item[C.] \(c > 0\)
    \item[D.] \(c = 0\)
\end{enumerate}
\end{frame}
\begin{frame}{Answer 1}
\small
\textbf{Correct Answer:} D
\end{frame}

% --- Question 2 ---
\begin{frame}{Question 2}
\small
As shown in the figure, \(O\) is the center of a circle with radius \(1\), and the circle intersects the coordinate axes at points \(A\) and \(B\). Point \(P\) lies on arc \(\overset{\frown}{AB}\) (not overlapping with \(A\) or \(B\)). Connect \(OP\), and let \(\angle POB = \alpha\). What are the coordinates of point \(P\)?

\begin{center}
    \includegraphics[width=0.42\linewidth]{hard_15_f2.png}
\end{center}
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] \((\sin\alpha,\ \sin\alpha)\)
    \item[B.] \((\cos\alpha,\ \cos\alpha)\)
    \item[C.] \((\cos\alpha,\ \sin\alpha)\)
    \item[D.] \((\sin\alpha,\ \cos\alpha)\)
\end{enumerate}
\end{frame}
\begin{frame}{Answer 2}
\small
\textbf{Correct Answer:} C
\end{frame}

% --- Question 3 ---
\begin{frame}{Question 3}
\small
The following table shows the age distribution of a school choir's members:

\begin{center}
\begin{tabular}{c|cccc}
Age (years) & \(13\) & \(14\) & \(15\) & \(16\) \\
\hline
Frequency & \(5\) & \(15\) & \(x\) & \(10 - x\) \\
\end{tabular}
\end{center}

For different values of \(x\), which pair of statistical measures regarding age will not change?
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] Mean, Median
    \item[B.] Mode, Median
    \item[C.] Mean, Standard Deviation
    \item[D.] Median, Standard Deviation
\end{enumerate}
\end{frame}
\begin{frame}{Answer 3}
\small
\textbf{Correct Answer:} B
\end{frame}

\begin{frame}{Question 4}

Given that the line \(y = kx + k + 1\) passes through the points \((m,\ n + 3)\) and \((m + 1,\ 2n - 1)\), and that \(0 < k < 2\), what is a possible value of \(n\)?
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] \(3\)
    \item[B.] \(4\)
    \item[C.] \(5\)
    \item[D.] \(6\)
\end{enumerate}
\end{frame}
\begin{frame}{Answer 4}
\small
\textbf{Correct Answer:} C
\end{frame}

% --- Question 5 ---
\begin{frame}{Question 5}
\small
The graph of the quadratic function \(y = |a|x^{2} + bx + c\) passes through the following five points:
\[
A(m,\ n),\quad B(0,\ y_{1}),\quad C(3 - m,\ n),\quad D(\sqrt{2},\ y_{2}),\quad E(2,\ y_{3}).
\]
What is the correct order of \(y_{1}\), \(y_{2}\), \(y_{3}\)?
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] \(y_{1} < y_{2} < y_{3}\)
    \item[B.] \(y_{1} < y_{3} < y_{2}\)
    \item[C.] \(y_{3} < y_{2} < y_{1}\)
    \item[D.] \(y_{2} < y_{3} < y_{1}\)
\end{enumerate}
\end{frame}
\begin{frame}{Answer 5}
\small
\textbf{Correct Answer:} B
\end{frame}

\begin{frame}{Question 6}

As shown in the figure, \(6\) identical rhombuses form a grid. The vertices of the rhombuses are called grid points. It is known that one angle of the rhombus, \(\angle O\), is \(60^\circ\), and points \(A\), \(B\), and \(C\) are located on grid points. Find the value of \(\tan\angle ABC\).

% --- Question 7 ---
\begin{frame}{Question 7}
\small
Given the quadratic function
\[
y = x^{2} - 2ax + a \quad (a \neq 0)
\]
passes through two points \(A\!\left(\dfrac{a}{2},\ y_{1}\right)\) and \(B(3a,\ y_{2})\), which of the following statements is correct?
\vspace{0.35cm}
\begin{enumerate}
    \item[A.] There exists a real number \(a\) such that \(y_{1} > a\).
    \item[B.] For all real numbers \(a\), \(y_{1} > a\) is always true.
    \item[C.] There exists a real number \(a\) such that \(y_{2} < 0\).
    \item[D.] For all real numbers \(a\), \(y_{2} < 0\) is always true.
\end{enumerate}
\end{frame}
\begin{frame}{Answer 7}
\small
\textbf{Correct Answer:} C
\end{frame}

Line \(AE \perp OC\), with foot of the perpendicular at point \(E\). The extension of line \(BE\) intersects \(AD\) at point \(F\). Find the value of \(\dfrac{OE}{AE}\).
